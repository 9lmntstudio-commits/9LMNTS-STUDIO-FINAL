                            <button onClick={()=>setView('god')} className={`px-4 py-1.5 rounded-md text-xs font-bold transition-all ${view==='god' ? 'bg-purple-600 text-white' : 'text-gray-500 hover:text-white'}`}>
                                <div className="flex items-center">
                                    <ShieldAlert className="w-3 h-3 mr-1" />
                                    God Mode
                                </div>
                            </button>
                        </div>
                    </div>

                    {/* CONTENT RENDER */}
                    {view === 'investor' && <InvestorView assets={assets} updateAsset={updateAsset} demoMode={demoMode} />}
                    {view === 'developer' && <DeveloperView assets={assets} />}
                    {view === 'god' && <GodModeView assets={assets} />}
                </div>
            );
        };

        ReactDOM.render(<App />, document.getElementById('root'));
    </script>
